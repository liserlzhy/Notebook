html 代码

<script>
$(document).ready(function () {
    $("#btn").on("click",function () {
        /*发送请求*/
        $("#result").text("请求数据中，请稍候...");
        $.get("Server.php",{name:$("#namevalue").val()},function (data) {
            $("#result").text(data);
        }).error(function(){
            $("#result").text("有问题");
        });
    })
});
</script>
<input type="text" id="namevalue"/>
<br/>
<button id="btn">Send</button>
