JQuery CSS

css()，addClass()，removeClass()，toggleClass()

实例

================ myCSS.css ================= 
.style1{width:50px;height:50px;background-color:red;}
.style2{width:100px;height:100px;background-color:blue;}
================ index.html ================= 
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <link rel="stylesheet" type="text/css" href="../../CSS/myCSS.css"/>
    <script src="../../JS/jquery-3.3.1.min.js"></script>
    <script>
        $(document).ready(function () {
            /*CSS 单个属性设置*/
          /*  $("div").css("width","100px");
            $("div").css("height","100px");
            $("div").css("border","1px blue solid");*/

            /*CSS 多个属性设置*/
            //$("div").css({width:"100px",height:"50px",backgroundColor:"yellow"});

            /*添加类*/
             $("div").addClass("style1");
            /*移除类*/
             //$("div").removeClass("style1");

            $("div").click(function () {
                $(this).toggleClass("style2"); //在addClass()，和removeClass()中来回切换
            });

        });
    </script>
</head>
<body>
    <div></div>
</body>
</html>
